# textdiff/comparator/tests.py
from django.test import TestCase
from comparator.algorithms.preprocessing import convert_to_phonetic, clean_word
from comparator.algorithms.alignment import align_words
from comparator.algorithms.comparator import TextComparator

class PreprocessingTests(TestCase):
    def test_convert_to_phonetic(self):
        result = convert_to_phonetic("سلام")
        self.assertIsInstance(result, str)
    
    def test_clean_word(self):
        word, removed = clean_word("سَلام")
        self.assertEqual(word, "سلام")

class AlignmentTests(TestCase):
    def test_align_words(self):
        words1 = ["سلام", "دنیا"]
        words2 = ["سلام", "جهان"]
        aligned1, aligned2, ops, cost = align_words(words1, words2)
        self.assertEqual(len(aligned1), len(aligned2))

class ComparatorTests(TestCase):
    def test_text_comparator(self):
        comparator = TextComparator("سلام", "سلام")
        h1, h2, cost, ops = comparator.compare_texts()
        self.assertEqual(cost, 0)
